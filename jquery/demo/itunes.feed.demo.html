jQuery plugin - iTunes.feed demo

Source:jquery.itunes.feed.js - humming-bird - Google Code

Example:json feed

$.iTunes.feed({
    output:'json',
    category:$.iTunes.categories['New Releases'],
    sf:$.iTunes.countries.japan,
    limit:5,
    genre:$.iTunes.genre['Rock']
  }, function(result){
    var
    feed = result.json,
    $div = $('<div>'),
    entries = feed.entries;
    
    $div
      .addClass('feed-result')
      .append('<h3>' + feed.description + ' - json feed' + '</h3>');
    
    for ( var i =0, j = entries.length ; i < j ; i++ )
      $div.append(entries[i].content);

    $div.appendTo('#json-feed');
});

Example:XML feed

$.iTunes.feed({
    output:'xml',
    category:'topalbums',
    sf:$.iTunes.countries.japan,
    limit:5,
    genre:$.iTunes.genre['All']
  }, function(result){
    var
    xml = result.xml,
    $div = $('<div>'),
    items = xml.getElementsByTagName('item');
    
    $div
      .addClass('feed-result')
      .append('<h3>' + $(xml.childNodes).filter('description:eq(0)').text() + 
             ' - XML feed' + '</h3>');
    
    for (var i = 0, j = items.length; i < j; i++)
      $div.append($(items[i].getElementsByTagName('content:encoded')).text());

    $div.appendTo('#xml-feed');
});
