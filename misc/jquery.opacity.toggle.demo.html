$('#toggle-btn').toggle(function(){
  $('#toggle-target').animate({opacity:0},{
    complete: function(){
      $('#opacity').text($(this).css('opacity'));
    }
  });
},function(){

      $('#opacity').text($(this).css('opacity'));
    }

});

  $('#toggle-target2').animate({opacity:"toggle"},{
    complete: function(){
      $('#opacity2').text($(this).css('opacity'));
    }
  });
},function(){
  $('#toggle-target2').animate({opacity:"toggle"},{
    complete: function(){

    }
