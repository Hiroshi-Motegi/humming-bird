jQuery plugin - overlay demo

当ページをクリックするとページ全体をレイヤー要素で被います。

Source : http://humming-bird.googlecode.com/svn/trunk/jquery/lib/jquery.overlay.js

IE6 対応確認用 select 要素

Usage:

$(document).click(function(e){
    
    if (e.button == 0) {
        $.overlay.show({
            css:{background:'url(../../tool/css/img/bg_cb_120x120.png)'},
            animateParams:{opacity:0.8}
        },function(){
            $('<p>')
                .attr('id', 'pdesc')
                .text('Press Key [Enter] or [Esc]. or Click')
                .positionCenter()
                .click(function(e){
                    e.stopPropagation();
                })
                .appendTo(document.body);
        });
        
        $.overlay.bind(function(){
            $.overlay.hide({
                animateOptions:{duration:800}
            },function(){
                $('#pdesc').remove();
            });
        });
    }
    
    return false;
    
});
